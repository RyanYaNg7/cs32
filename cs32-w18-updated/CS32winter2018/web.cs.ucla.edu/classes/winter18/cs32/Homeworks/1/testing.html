bool isPermutation(KeyType a[], const KeyType b[], int n)
{
	for (int k = 0; k < n; k++)
	{
		int j;
		for (j = k; j < n; j++)
		{
			if (b[k] == a[j])
			{
				KeyType t = a[k];
				a[k] = a[j];
				a[j] = t;
				break;
			}
		}
		if (j == n)
			return false;
	}
	return true;
}

bool has(const Map& m, const KeyType& k, const ValueType& v)
{
	ValueType v2 = DEFAULTV;
	m.get(k, v2);
	ValueType v3 = DUMMYV;
	m.get(k, v3);
	return v2 == v  &&  v3 == v;
}
